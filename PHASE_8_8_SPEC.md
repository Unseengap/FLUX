```

                     │      FLUX WAVE SPACE            │
                     │   (432-dim semantic manifold)   │
                     │                                 │
                     │   ┌─────────────────────────┐   │
                     │   │   Field + GR + TL + CGN │   │
                     │   │   (modality-agnostic)   │   │
                     │   └─────────────────────────┘   │
                     │                                 │
                     └─────────────────────────────────┘
                              ▲             │

              │                                               │
    ┌─────────┴─────────┐                       ┌─────────────┴─────────┐

              │                                           │
    ┌─────────┼─────────┬─────────┐           ┌───────────┼─────────┬─────────┐
    │         │         │         │           │           │         │         │
    ▼         ▼         ▼         ▼           ▼           ▼         ▼         ▼
  Text     Grids    Images    Audio        Text       Grids     Images    Audio
```

The FLUX core doesn't know or care what modality it's processing.
It operates on 432-dimensional semantic waves. The adapters translate.

## Component 1: WaveToX Abstract Base + Registry

```python
"""
wave_to_x.py — The Universal Modality Interface

Every modality adapter implements this interface.
The FLUX core never imports specific adapters — it uses the registry.
"""

import torch
import torch.nn as nn
from torch import Tensor
from typing import Dict, Any, Optional, Type, Union
from abc import ABC, abstractmethod
from dataclasses import dataclass

# ─────────────────────────────────────────────
# Data Types
# ─────────────────────────────────────────────

@dataclass
class ModalitySpec:
    """Specification for a modality."""
    name: str                    # 'text', 'grid', 'image', 'audio'
    wave_dim: int = 432          # FLUX wave dimension (constant)
    supports_sequence: bool = True   # Can handle variable-length input?
    discrete: bool = False       # Discrete tokens (grid/text) vs continuous
    spatial: bool = False        # Has 2D/3D structure (grid/image)
    temporal: bool = False       # Has time dimension (audio/video)


MODALITY_SPECS = {
    'text': ModalitySpec('text', supports_sequence=True, discrete=True),
    'grid': ModalitySpec('grid', supports_sequence=False, discrete=True, spatial=True),
    'image': ModalitySpec('image', supports_sequence=True, discrete=False, spatial=True),
    'audio': ModalitySpec('audio', supports_sequence=True, discrete=False, temporal=True),
}


# ─────────────────────────────────────────────
# Abstract Base Classes
# ─────────────────────────────────────────────

class XToWave(nn.Module, ABC):
    """
    Input adapter: Modality → Wave space.
    
    Converts any input modality into FLUX's 432-dim semantic waves.
    The output can be:
    - [432] single vector (for grid-as-whole)
    - [seq, 432] sequence (for text, patches)
    """
    
    def __init__(self, modality: str, wave_dim: int = 432):
        super().__init__()
        self.modality = modality
        self.wave_dim = wave_dim
        self.spec = MODALITY_SPECS.get(modality, ModalitySpec(modality))
    
    @abstractmethod
    def encode(self, x: Any) -> Tensor:
        """
        Encode input to wave space.
        
        Args:
            x: Input in native format (str, grid tensor, image tensor, etc.)
        
        Returns:
            Tensor of shape [432] or [seq, 432]
        """
        pass
    
    def forward(self, x: Any) -> Tensor:
        return self.encode(x)


class WaveToX(nn.Module, ABC):
    """
    Output adapter: Wave space → Modality.
    
    Converts FLUX's 432-dim semantic waves into any output modality.
    The input can be:
    - [432] single vector
    - [seq, 432] sequence of waves
    """
    
    def __init__(self, modality: str, wave_dim: int = 432):
        super().__init__()
        self.modality = modality
        self.wave_dim = wave_dim
        self.spec = MODALITY_SPECS.get(modality, ModalitySpec(modality))
    
    @abstractmethod
    def decode(self, waves: Tensor, **kwargs) -> Any:
        """
        Decode waves to output modality.
        
        Args:
            waves: [432] or [seq, 432] wave tensor
            **kwargs: Modality-specific options (temperature, grid_size, etc.)
        
        Returns:
            Output in native format (str, grid tensor, image tensor, etc.)
        """
        pass
    
    def forward(self, waves: Tensor, **kwargs) -> Any:
        return self.decode(waves, **kwargs)


# ─────────────────────────────────────────────
# Registry
# ─────────────────────────────────────────────

_INPUT_ADAPTERS: Dict[str, Type[XToWave]] = {}
_OUTPUT_ADAPTERS: Dict[str, Type[WaveToX]] = {}


def register_input_adapter(modality: str):
    """Decorator to register an input adapter."""
    def decorator(cls: Type[XToWave]):
        _INPUT_ADAPTERS[modality] = cls
        return cls
    return decorator


def register_output_adapter(modality: str):
    """Decorator to register an output adapter."""
    def decorator(cls: Type[WaveToX]):
        _OUTPUT_ADAPTERS[modality] = cls
        return cls
    return decorator


def get_input_adapter(modality: str, **kwargs) -> XToWave:
    """Get input adapter for modality."""
    if modality not in _INPUT_ADAPTERS:
        raise ValueError(f"No input adapter for modality: {modality}")
    return _INPUT_ADAPTERS[modality](**kwargs)


def get_output_adapter(modality: str, **kwargs) -> WaveToX:
    """Get output adapter for modality."""
    if modality not in _OUTPUT_ADAPTERS:
        raise ValueError(f"No output adapter for modality: {modality}")
    return _OUTPUT_ADAPTERS[modality](**kwargs)


def list_adapters() -> Dict[str, Dict[str, bool]]:
    """List all registered adapters."""
    return {
        'input': {m: True for m in _INPUT_ADAPTERS},
        'output': {m: True for m in _OUTPUT_ADAPTERS},
    }
```

```python
"""
text_adapters.py — Text ↔ Wave adapters

TextToWave wraps the Phase 1 CSE (already does this perfectly).
WaveToText is the last-mile byte decoder (simpler than Phase 8's full decoder).
"""

import torch
import torch.nn as nn
import torch.nn.functional as F
from torch import Tensor
from typing import Optional
import sys
from pathlib import Path

ROOT = Path(__file__).parent.parent.parent
sys.path.insert(0, str(ROOT))
sys.path.insert(0, str(ROOT / 'phases' / 'phase1'))

from wave_to_x import XToWave, WaveToX, register_input_adapter, register_output_adapter


@register_input_adapter('text')
class TextToWave(XToWave):
    """
    Text → Wave adapter.
    
    Wraps the Phase 1 CSE. This is the original FLUX encoder.
    No new architecture needed — CSE already does this perfectly.
    """
    
    def __init__(
        self,
        wave_dim: int = 432,
        cse: Optional[nn.Module] = None,
        device: str = 'cpu',
    ):
        super().__init__('text', wave_dim)
        self.device = device
        
        if cse is not None:
            self.cse = cse
        else:
            # Load from checkpoint
            from flux_utils import load_checkpoint
            from cse import ContinuousSemanticEncoder
            ckpt = load_checkpoint(1)
            self.cse = ContinuousSemanticEncoder(**ckpt['config'])
            self.cse.load_state_dict(ckpt['state_dict'])
            self.cse.eval()
        
        self.cse = self.cse.to(device)
    
    def encode(self, text: str) -> Tensor:
        """
        Encode text to wave sequence.
        
        Args:
            text: UTF-8 string
        
        Returns:
            [seq_len, 432] wave tensor
        """
        with torch.no_grad():
            wave = self.cse.encode(text)
        return wave.full.to(self.device)
    
    def encode_pooled(self, text: str) -> Tensor:
        """
        Encode text to single wave vector (mean pooled).
        
        Returns:
            [432] wave tensor
        """
        return self.encode(text).mean(dim=0)


@register_output_adapter('text')
class WaveToText(WaveToX):
    """
    Wave → Text adapter.
    
    Decodes a single wave (representing ~1 word/chunk) to UTF-8 bytes.
    Simpler than Phase 8's full decoder — this handles one chunk at a time.
    
    Architecture:
        wave [432] → hidden [256] → logits [256 × max_bytes]
        Autoregressive byte-by-byte with learned stop token.
    """
    
    def __init__(
        self,
        wave_dim: int = 432,
        hidden_dim: int = 256,
        max_bytes: int = 24,
        device: str = 'cpu',
    ):
        super().__init__('text', wave_dim)
        self.hidden_dim = hidden_dim
        self.max_bytes = max_bytes
        self.device = device
        
        # Wave → hidden
        self.wave_proj = nn.Linear(wave_dim, hidden_dim)
        
        # Byte embedding (256 possible bytes + stop token)
        self.byte_embed = nn.Embedding(257, hidden_dim)
        
        # GRU for autoregressive decoding
        self.gru = nn.GRU(hidden_dim, hidden_dim, batch_first=True)
        
        # Output projection
        self.out_proj = nn.Linear(hidden_dim, 257)
        
        self.to(device)
    
    def decode(
        self,
        wave: Tensor,
        temperature: float = 0.8,
        max_bytes: Optional[int] = None,
    ) -> str:
        """
        Decode wave to text string.
        
        Args:
            wave: [432] wave tensor
            temperature: Sampling temperature
            max_bytes: Override max bytes
        
        Returns:
            Decoded UTF-8 string
        """
        max_len = max_bytes or self.max_bytes
        
        # Project wave to hidden
        wave = wave.to(self.device)
        if wave.dim() == 1:
            wave = wave.unsqueeze(0)
        
        hidden = self.wave_proj(wave).unsqueeze(0)  # [1, 1, hidden]
        
        # Start token (space = 32)
        current = torch.tensor([[32]], device=self.device)
        
        bytes_out = []
        for _ in range(max_len):
            # Embed current byte
            emb = self.byte_embed(current)  # [1, 1, hidden]
            
            # GRU step
            out, hidden = self.gru(emb, hidden)
            
            # Predict next byte
            logits = self.out_proj(out.squeeze(1))  # [1, 257]
            
            # Sample
            if temperature > 0:
                probs = F.softmax(logits / temperature, dim=-1)
                next_byte = torch.multinomial(probs, 1)
            else:
                next_byte = logits.argmax(dim=-1, keepdim=True)
            
            byte_val = next_byte.item()
            
            # Stop token (256) or null byte
            if byte_val >= 256 or byte_val == 0:
                break
            
            bytes_out.append(byte_val)
            current = next_byte
        
        # Decode bytes to string
        return bytes(bytes_out).decode('utf-8', errors='replace')
    
    def forward_train(
        self,
        wave: Tensor,
        target_bytes: Tensor,
    ) -> Tensor:
        """
        Training forward pass.
        
        Args:
            wave: [batch, 432] wave tensors
            target_bytes: [batch, max_bytes] target byte sequences
        
        Returns:
            [batch, max_bytes, 257] logits
        """
        # Project waves
        hidden = self.wave_proj(wave).unsqueeze(0)  # [1, batch, hidden]
        
        # Embed target (shifted by 1 for teacher forcing)
        emb = self.byte_embed(target_bytes)  # [batch, seq, hidden]
        
        # GRU
        out, _ = self.gru(emb, hidden)
        
        # Project to logits
        logits = self.out_proj(out)  # [batch, seq, 257]
        
        return logits
```

## Component 3: Grid Adapters (ARC-Ready)

```python
"""
grid_adapters.py — Grid ↔ Wave adapters

The key to ARC Prize. Encodes 2D grids (up to 30×30, 10 colors)
into FLUX's wave space, enabling spatial reasoning.

Two encoding modes:
1. Holistic: Entire grid → single [432] vector (for rule matching)
2. Spatial: Grid → [H×W, 432] per-cell waves (for detailed reasoning)
"""

import torch
import torch.nn as nn
import torch.nn.functional as F
from torch import Tensor
from typing import Optional, Tuple, Union
import math

from wave_to_x import XToWave, WaveToX, register_input_adapter, register_output_adapter


@register_input_adapter('grid')
class GridToWave(XToWave):
    """
    Grid → Wave adapter.
    
    Encodes ARC-style grids (H×W matrix with integer colors 0-9)
    into FLUX's 432-dimensional wave space.
    
    Architecture:
        1. Color embedding: each color → learned 32-dim vector
        2. Position encoding: (x, y) → 64-dim spatial signal
        3. Local convolution: capture 3×3 neighborhood patterns
        4. Global pooling: compress to single wave (optional)
    
    The encoding captures:
        - What colors are present
        - Where they are located
        - What patterns they form (lines, blocks, symmetry)
    """
    
    def __init__(
        self,
        wave_dim: int = 432,
        num_colors: int = 10,
        max_size: int = 30,
        color_dim: int = 32,
        pattern_dim: int = 64,
        device: str = 'cpu',
    ):
        super().__init__('grid', wave_dim)
        self.num_colors = num_colors
        self.max_size = max_size
        self.color_dim = color_dim
        self.pattern_dim = pattern_dim
        self.device = device
        
        # Color embedding (0-9 → 32-dim)
        self.color_embed = nn.Embedding(num_colors + 1, color_dim)  # +1 for padding
        
        # Position encoding (learnable, max 30×30)
        self.pos_embed_h = nn.Embedding(max_size, pattern_dim // 2)
        self.pos_embed_w = nn.Embedding(max_size, pattern_dim // 2)
        
        # Local pattern convolutions
        self.conv1 = nn.Conv2d(color_dim, pattern_dim, 3, padding=1)
        self.conv2 = nn.Conv2d(pattern_dim, pattern_dim, 3, padding=1)
        self.conv3 = nn.Conv2d(pattern_dim, pattern_dim, 3, padding=1)
        
        # Global pooling → wave projection
        # color_dim + pattern_dim (position) + pattern_dim (conv) = feature_dim
        feature_dim = color_dim + pattern_dim + pattern_dim
        self.to_wave = nn.Linear(feature_dim, wave_dim)
        
        # For holistic mode: compress entire grid to one vector
        self.global_pool = nn.Sequential(
            nn.AdaptiveAvgPool2d(1),
            nn.Flatten(),
            nn.Linear(pattern_dim, wave_dim),
        )
        
        self.to(device)
    
    def encode(
        self,
        grid: Union[Tensor, list],
        mode: str = 'holistic',
    ) -> Tensor:
        """
        Encode grid to wave space.
        
        Args:
            grid: [H, W] integer tensor (values 0-9) or list of lists
            mode: 'holistic' (single vector) or 'spatial' (per-cell)
        
        Returns:
            'holistic': [432] wave tensor
            'spatial': [H*W, 432] wave tensor
        """
        # Convert to tensor if needed
        if isinstance(grid, list):
            grid = torch.tensor(grid, dtype=torch.long, device=self.device)
        else:
            grid = grid.to(self.device)
        
        H, W = grid.shape
        
        # Color embedding: [H, W] → [H, W, color_dim]
        color_feats = self.color_embed(grid)
        
        # Position encoding
        h_pos = self.pos_embed_h(torch.arange(H, device=self.device))  # [H, 32]
        w_pos = self.pos_embed_w(torch.arange(W, device=self.device))  # [W, 32]
        
        # Expand to grid: [H, W, 64]
        pos_feats = torch.cat([
            h_pos.unsqueeze(1).expand(-1, W, -1),
            w_pos.unsqueeze(0).expand(H, -1, -1),
        ], dim=-1)
        
        # Convolution for local patterns
        # [H, W, color_dim] → [1, color_dim, H, W]
        x = color_feats.permute(2, 0, 1).unsqueeze(0)
        x = F.relu(self.conv1(x))
        x = F.relu(self.conv2(x))
        x = F.relu(self.conv3(x))
        # → [1, pattern_dim, H, W] → [H, W, pattern_dim]
        pattern_feats = x.squeeze(0).permute(1, 2, 0)
        
        if mode == 'holistic':
            # Global pooling over entire grid
            global_feat = self.global_pool(x)  # [1, wave_dim]
            return global_feat.squeeze(0)
        
        else:  # spatial
            # Concatenate features per cell
            cell_feats = torch.cat([color_feats, pos_feats, pattern_feats], dim=-1)
            # [H, W, feature_dim] → [H*W, feature_dim]
            cell_feats = cell_feats.view(H * W, -1)
            # Project to wave space
            waves = self.to_wave(cell_feats)  # [H*W, 432]
            return waves
    
    def encode_pair(
        self,
        input_grid: Union[Tensor, list],
        output_grid: Union[Tensor, list],
    ) -> Tuple[Tensor, Tensor, Tensor]:
        """
        Encode input/output pair and compute delta.
        
        This is the key for ARC: the delta represents the transformation rule.
        
        Returns:
            (input_wave, output_wave, delta_wave)
        """
        input_wave = self.encode(input_grid, mode='holistic')
        output_wave = self.encode(output_grid, mode='holistic')
        delta_wave = output_wave - input_wave
        return input_wave, output_wave, delta_wave


@register_output_adapter('grid')
class WaveToGrid(WaveToX):
    """
    Wave → Grid adapter.
    
    Decodes FLUX waves back to ARC-style grids.
    
    Two modes:
    1. From delta: Apply transformation wave to input grid
    2. Direct: Decode wave directly to grid (for generation)
    
    Architecture:
        wave [432] → hidden [256] → [H, W, num_colors] logits → argmax
    """
    
    def __init__(
        self,
        wave_dim: int = 432,
        num_colors: int = 10,
        max_size: int = 30,
        hidden_dim: int = 256,
        device: str = 'cpu',
    ):
        super().__init__('grid', wave_dim)
        self.num_colors = num_colors
        self.max_size = max_size
        self.hidden_dim = hidden_dim
        self.device = device
        
        # Wave → spatial features
        self.wave_proj = nn.Linear(wave_dim, hidden_dim)
        
        # Size predictor (predicts H, W)
        self.size_head = nn.Sequential(
            nn.Linear(wave_dim, 64),
            nn.ReLU(),
            nn.Linear(64, 2),  # [H, W] as continuous, then round
        )
        
        # Spatial decoder
        self.spatial_expand = nn.Linear(hidden_dim, max_size * max_size * hidden_dim // 4)
        
        self.decoder = nn.Sequential(
            nn.ConvTranspose2d(hidden_dim // 4, hidden_dim // 2, 3, padding=1),
            nn.ReLU(),
            nn.ConvTranspose2d(hidden_dim // 2, num_colors, 3, padding=1),
        )
        
        # For transform mode: combine input + delta
        self.transform_combine = nn.Sequential(
            nn.Linear(wave_dim * 2, hidden_dim),
            nn.ReLU(),
            nn.Linear(hidden_dim, wave_dim),
        )
        
        self.to(device)
    
    def decode(
        self,
        wave: Tensor,
        grid_size: Optional[Tuple[int, int]] = None,
        input_grid: Optional[Tensor] = None,
        temperature: float = 0.0,
    ) -> Tensor:
        """
        Decode wave to grid.
        
        Args:
            wave: [432] wave tensor (the transformation or direct encoding)
            grid_size: (H, W) if known, else predicted
            input_grid: For transform mode, the input to transform
            temperature: >0 for sampling, 0 for argmax
        
        Returns:
            [H, W] integer tensor (values 0-9)
        """
        wave = wave.to(self.device)
        if wave.dim() == 1:
            wave = wave.unsqueeze(0)
        
        # If input_grid provided, this is transform mode
        if input_grid is not None:
            # Combine input encoding with delta wave
            from wave_to_x import get_input_adapter
            grid_encoder = get_input_adapter('grid', device=self.device)
            input_wave = grid_encoder.encode(input_grid, mode='holistic')
            
            combined = torch.cat([input_wave.unsqueeze(0), wave], dim=-1)
            wave = self.transform_combine(combined)
        
        # Predict or use given size
        if grid_size is None:
            size_pred = self.size_head(wave)
            H = int(torch.clamp(size_pred[0, 0], 1, self.max_size).item())
            W = int(torch.clamp(size_pred[0, 1], 1, self.max_size).item())
        else:
            H, W = grid_size
        
        # Decode to grid
        hidden = self.wave_proj(wave)  # [1, hidden]
        
        # Expand to spatial
        spatial = self.spatial_expand(hidden)  # [1, max*max*hidden//4]
        spatial = spatial.view(1, self.hidden_dim // 4, self.max_size, self.max_size)
        
        # Decode
        logits = self.decoder(spatial)  # [1, num_colors, max, max]
        
        # Crop to actual size
        logits = logits[:, :, :H, :W]  # [1, num_colors, H, W]
        
        # Sample or argmax
        if temperature > 0:
            probs = F.softmax(logits / temperature, dim=1)
            # Sample per cell
            probs_flat = probs.permute(0, 2, 3, 1).reshape(-1, self.num_colors)
            samples = torch.multinomial(probs_flat, 1).squeeze(-1)
            grid = samples.view(H, W)
        else:
            grid = logits.argmax(dim=1).squeeze(0)  # [H, W]
        
        return grid
    
    def apply_transform(
        self,
        delta_wave: Tensor,
        input_grid: Union[Tensor, list],
        grid_size: Optional[Tuple[int, int]] = None,
    ) -> Tensor:
        """
        Apply a transformation wave to an input grid.
        
        This is the key ARC operation:
            output = apply(delta, input)
        
        Where delta was extracted from training examples.
        """
        return self.decode(
            wave=delta_wave,
            grid_size=grid_size,
            input_grid=input_grid,
            temperature=0.0,
        )
```

## Component 4: Image Adapters (Multi-Physics Universal Renderer)

```python
"""
image_adapters.py — Image ↔ Wave adapters

WaveToImage_Universal: Three physics engines blended for unlimited style.
- Gravity: Mass-based influence (realistic, structured)
- Interference: Wave physics (artistic, patterns)
- Thermodynamic: Energy settling (abstract, emergent)

Style = blend weights. Same waves, different physics, different images.
"""

import torch
import torch.nn as nn
import torch.nn.functional as F
from torch import Tensor
from typing import Dict, Optional, Tuple
import math

from wave_to_x import XToWave, WaveToX, register_input_adapter, register_output_adapter


@register_input_adapter('image')
class ImageToWave(XToWave):
    """
    Image → Wave adapter.
    
    Encodes images as sequences of patch waves.
    Each patch → one wave preserving spatial semantics.
    """
    
    def __init__(
        self,
        wave_dim: int = 432,
        patch_size: int = 16,
        image_size: int = 256,
        device: str = 'cpu',
    ):
        super().__init__('image', wave_dim)
        self.patch_size = patch_size
        self.image_size = image_size
        self.device = device
        
        patches_per_side = image_size // patch_size
        self.num_patches = patches_per_side ** 2
        
        # Patch → wave projection
        patch_dim = patch_size * patch_size * 3
        self.patch_to_wave = nn.Sequential(
            nn.Linear(patch_dim, 256),
            nn.ReLU(),
            nn.Linear(256, wave_dim),
        )
        
        # Learnable position encoding per patch
        self.pos_embed = nn.Parameter(torch.randn(self.num_patches, wave_dim) * 0.02)
        
        self.to(device)
    
    def encode(self, image: Tensor) -> Tensor:
        """
        Encode image to wave sequence.
        
        Args:
            image: [3, H, W] or [H, W, 3] RGB tensor
        
        Returns:
            [num_patches, 432] wave tensor
        """
        if image.shape[-1] == 3:  # HWC → CHW
            image = image.permute(2, 0, 1)
        
        image = image.to(self.device)
        C, H, W = image.shape
        
        # Resize if needed
        if H != self.image_size or W != self.image_size:
            image = F.interpolate(
                image.unsqueeze(0), 
                size=(self.image_size, self.image_size),
                mode='bilinear'
            ).squeeze(0)
        
        # Extract patches
        patches = image.unfold(1, self.patch_size, self.patch_size)
        patches = patches.unfold(2, self.patch_size, self.patch_size)
        patches = patches.contiguous().view(C, -1, self.patch_size, self.patch_size)
        patches = patches.permute(1, 0, 2, 3)  # [num_patches, C, ps, ps]
        patches = patches.reshape(self.num_patches, -1)  # [num_patches, C*ps*ps]
        
        # Project to wave space
        waves = self.patch_to_wave(patches)  # [num_patches, 432]
        
        # Add position encoding
        waves = waves + self.pos_embed
        
        return waves
    
    def encode_pooled(self, image: Tensor) -> Tensor:
        """Encode image to single wave vector."""
        return self.encode(image).mean(dim=0)


@register_output_adapter('image')
class WaveToImage_Universal(WaveToX):
    """
    Universal image decoder — realistic, artistic, abstract, anything.
    
    Core insight: The FIELD stores semantics. The RENDERER chooses physics.
    
    Three rendering paths that blend:
    - Gravity: Mass-based influence (realistic, structured)
    - Interference: Wave physics (artistic, patterns)  
    - Thermodynamic: Energy settling (abstract, emergent)
    
    Style = blend weights. "realistic" = more gravity. "abstract" = more thermo.
    OR: Let the waves themselves determine the blend (self-routing).
    """
    
    def __init__(
        self,
        wave_dim: int = 432,
        image_size: int = 256,
        device: str = 'cpu',
    ):
        super().__init__('image', wave_dim)
        self.image_size = image_size
        self.device = device
        
        # ─────────────────────────────────────────
        # Shared: Wave → Spatial Properties
        # ─────────────────────────────────────────
        self.wave_to_position = nn.Linear(wave_dim, 2)
        self.wave_to_color = nn.Linear(wave_dim, 3)
        self.wave_to_scale = nn.Linear(wave_dim, 1)
        
        # ─────────────────────────────────────────
        # Gravity Path (Realistic)
        # ─────────────────────────────────────────
        self.wave_to_mass = nn.Linear(wave_dim, 1)
        self.wave_to_texture = nn.Linear(wave_dim, 64)
        self.texture_decoder = nn.Sequential(
            nn.Linear(64, 128),
            nn.ReLU(),
            nn.Linear(128, 8 * 8 * 3),
        )
        
        # ─────────────────────────────────────────
        # Interference Path (Artistic)
        # ─────────────────────────────────────────
        self.wave_to_phase = nn.Linear(wave_dim, 1)
        self.wave_to_frequency = nn.Linear(wave_dim, 3)  # Per-channel freq
        self.wave_to_amplitude = nn.Linear(wave_dim, 3)
        
        # ─────────────────────────────────────────
        # Thermodynamic Path (Abstract)
        # ─────────────────────────────────────────
        self.wave_to_energy_target = nn.Linear(wave_dim, 16)
        self.energy_to_rgb = nn.Linear(16, 3)
        
        # ─────────────────────────────────────────
        # Style Router (wave-determined or explicit)
        # ─────────────────────────────────────────
        self.wave_to_style = nn.Linear(wave_dim, 3)
        
        # Learnable style anchors
        self.style_realistic = nn.Parameter(torch.tensor([0.7, 0.2, 0.1]))
        self.style_artistic = nn.Parameter(torch.tensor([0.2, 0.7, 0.1]))
        self.style_abstract = nn.Parameter(torch.tensor([0.1, 0.3, 0.6]))
        self.style_photographic = nn.Parameter(torch.tensor([0.9, 0.05, 0.05]))
        self.style_psychedelic = nn.Parameter(torch.tensor([0.1, 0.6, 0.3]))
        
        self.to(device)
    
    def _render_gravity(self, waves: Tensor, coords: Tensor) -> Tensor:
        """Realistic: Mass attracts, textures fill regions."""
        H = W = self.image_size
        n = waves.shape[0]
        device = waves.device
        
        positions = torch.sigmoid(self.wave_to_position(waves)) * H
        masses = F.softplus(self.wave_to_mass(waves))
        colors = torch.sigmoid(self.wave_to_color(waves))
        textures = self.wave_to_texture(waves)
        
        image = torch.zeros(3, H, W, device=device)
        weights = torch.zeros(H, W, device=device)
        
        for i in range(n):
            dist = torch.norm(coords - positions[i], dim=-1) + 1e-4
            influence = masses[i] / (dist ** 2)
            
            # Base color with distance falloff
            contribution = colors[i].view(3, 1, 1) * influence
            
            # Add texture variation
            tex_patch = self.texture_decoder(textures[i]).view(3, 8, 8)
            tex_patch = F.interpolate(
                tex_patch.unsqueeze(0), size=(H, W), mode='bilinear'
            )[0]
            contribution = contribution * (1 + 0.1 * tex_patch)
            
            image += contribution
            weights += influence
        
        return image / (weights.unsqueeze(0) + 1e-8)
    
    def _render_interference(self, waves: Tensor, coords: Tensor) -> Tensor:
        """Artistic: Wave interference patterns."""
        H = W = self.image_size
        device = waves.device
        
        positions = torch.sigmoid(self.wave_to_position(waves)) * H
        phases = self.wave_to_phase(waves)
        freqs = torch.sigmoid(self.wave_to_frequency(waves)) * 0.5 + 0.05
        amps = torch.sigmoid(self.wave_to_amplitude(waves))
        
        image = torch.zeros(3, H, W, device=device)
        
        for i in range(waves.shape[0]):
            dist = torch.norm(coords - positions[i], dim=-1)
            
            # Per-channel interference (creates color shifts)
            for c in range(3):
                wave_pattern = amps[i, c] * torch.cos(
                    2 * math.pi * freqs[i, c] * dist + phases[i]
                )
                decay = torch.exp(-dist / (H / 3))
                image[c] += wave_pattern * decay
        
        # Normalize with artistic contrast
        image = torch.tanh(image * 2) * 0.5 + 0.5
        return image
    
    def _render_thermodynamic(self, waves: Tensor, coords: Tensor) -> Tensor:
        """Abstract: Energy minimization creates emergent structure."""
        H = W = self.image_size
        device = waves.device
        
        energy_targets = torch.tanh(self.wave_to_energy_target(waves))
        positions = torch.sigmoid(self.wave_to_position(waves)) * H
        scales = torch.sigmoid(self.wave_to_scale(waves)) * (H / 4) + 10
        
        # Initialize with structured noise
        x, y = coords[..., 0], coords[..., 1]
        base_pattern = torch.sin(x / 20) * torch.cos(y / 20)
        image = base_pattern.unsqueeze(0).expand(3, -1, -1).clone()
        
        for i in range(waves.shape[0]):
            dist = torch.norm(coords - positions[i], dim=-1)
            influence = torch.exp(-dist**2 / (2 * scales[i]**2))
            target_rgb = torch.sigmoid(self.energy_to_rgb(energy_targets[i]))
            
            for c in range(3):
                image[c] = image[c] * (1 - influence) + target_rgb[c] * influence
        
        # Add entropy based on wave intensity variance
        intensity_var = waves[:, -16:].var()
        if intensity_var > 0.1:
            noise_scale = float(intensity_var) * 0.2
            image = image + torch.randn_like(image) * noise_scale
        
        return torch.clamp(image, 0, 1)
    
    def decode(
        self,
        waves: Tensor,
        style: str = 'auto',
        style_weights: Optional[Tensor] = None,
    ) -> Tensor:
        """
        Decode waves to image with style control.
        
        Args:
            waves: [N, 432] semantic waves
            style: 'auto', 'realistic', 'artistic', 'abstract', 
                   'photographic', 'psychedelic'
            style_weights: [3] tensor for explicit blend
        
        Returns:
            [3, H, W] RGB image
        """
        H = W = self.image_size
        device = waves.device
        
        # Build coordinate grid
        y, x = torch.meshgrid(
            torch.arange(H, device=device, dtype=torch.float),
            torch.arange(W, device=device, dtype=torch.float),
            indexing='ij'
        )
        coords = torch.stack([x, y], dim=-1)
        
        # Determine style weights
        if style_weights is not None:
            weights = F.softmax(style_weights, dim=0)
        elif style == 'auto':
            wave_style = self.wave_to_style(waves.mean(dim=0))
            weights = F.softmax(wave_style, dim=0)
        elif style == 'realistic':
            weights = F.softmax(self.style_realistic, dim=0)
        elif style == 'artistic':
            weights = F.softmax(self.style_artistic, dim=0)
        elif style == 'abstract':
            weights = F.softmax(self.style_abstract, dim=0)
        elif style == 'photographic':
            weights = F.softmax(self.style_photographic, dim=0)
        elif style == 'psychedelic':
            weights = F.softmax(self.style_psychedelic, dim=0)
        else:
            weights = torch.tensor([0.33, 0.33, 0.34], device=device)
        
        # Render all three paths
        img_gravity = self._render_gravity(waves, coords)
        img_interference = self._render_interference(waves, coords)
        img_thermo = self._render_thermodynamic(waves, coords)
        
        # Blend according to style weights
        image = (
            weights[0] * img_gravity +
            weights[1] * img_interference +
            weights[2] * img_thermo
        )
        
        return torch.clamp(image, 0, 1)
    
    def decode_multi_style(self, waves: Tensor) -> Dict[str, Tensor]:
        """Generate all styles at once for comparison."""
        return {
            'realistic': self.decode(waves, style='realistic'),
            'artistic': self.decode(waves, style='artistic'),
            'abstract': self.decode(waves, style='abstract'),
            'photographic': self.decode(waves, style='photographic'),
            'psychedelic': self.decode(waves, style='psychedelic'),
            'auto': self.decode(waves, style='auto'),
        }
```

## Component 5: FluxToAny — The Universal Model

```python
"""
flux_to_any.py — Universal modality model

FluxToAny extends Flux-beta with pluggable input/output adapters.
The core FLUX reasoning (Field, GR, TL, CGN, Memory) is unchanged.
Only the I/O layer is modular.
"""

import torch
import torch.nn as nn
from torch import Tensor
from pathlib import Path
from typing import Dict, Any, Optional, Union, Tuple
import sys

ROOT = Path(__file__).parent.parent.parent
sys.path.insert(0, str(ROOT))

from flux_utils import get_device
from wave_to_x import (
    XToWave, WaveToX,
    get_input_adapter, get_output_adapter, list_adapters,
)


class FluxToAny(nn.Module):
    """
    Universal FLUX model with pluggable modality adapters.
    
    Core components (from Flux-beta.flx):
        - CSE (wrapped as TextToWave)
        - Field + GR + TL + CGN
        - Memory (Episodic, Semantic, Working)
    
    I/O Adapters (pluggable):
        - Input: TextToWave, GridToWave, ImageToWave, AudioToWave
        - Output: WaveToText, WaveToGrid, WaveToImage, WaveToAudio
    
    Usage:
        model = FluxToAny.from_flx('checkpoints/Flux-beta.flx')
        
        # Text → Text (original FLUX)
        output = model.process("Hello world", output_modality='text')
        
        # Grid → Grid (ARC mode)
        output = model.process(arc_grid, input_modality='grid', output_modality='grid')
        
        # Text → Grid (cross-modal)
        output = model.process("A 3x3 red square", output_modality='grid')
    """
    
    def __init__(
        self,
        flx_path: Optional[Path] = None,
        device: str = 'cpu',
    ):
        super().__init__()
        self.device = device
        
        # Load core from .flx
        if flx_path:
            self._load_flx(flx_path)
        else:
            self._init_empty()
        
        # Initialize adapters
        self.input_adapters: Dict[str, XToWave] = {}
        self.output_adapters: Dict[str, WaveToX] = {}
        
        # Always have text adapter (wraps CSE)
        self._init_text_adapters()
    
    def _load_flx(self, path: Path):
        """Load core components from .flx file."""
        flx = torch.load(path, map_location='cpu')
        self.flx_version = flx.get('version', '1.0-beta')
        
        # Load CSE
        from phases.phase1.cse import ContinuousSemanticEncoder
        cse_cfg = flx['cse']['config']
        self.cse = ContinuousSemanticEncoder(**cse_cfg)
        self.cse.load_state_dict(flx['cse']['state_dict'])
        self.cse.eval()
        
        # Load Field (Phase 2)
        if 'field' in flx:
            from phases.phase2.field import ResonanceField
            field_cfg = flx['field'].get('config', flx['field'].get('field_config', {}))
            self.field = ResonanceField(**field_cfg)
            self.field.load_state_dict(flx['field'].get('state_dict', flx['field'].get('field_state', {})))
        
        # Load Decoder (Phase 8) — becomes WaveToText backbone
        if 'decoder' in flx:
            self.byte_decoder_state = flx['decoder']
        
        # Load memory
        if 'memory' in flx:
            self.memory_state = flx['memory']
        
        # Move to device
        self.cse = self.cse.to(self.device)
        if hasattr(self, 'field'):
            self.field = self.field.to(self.device)
        
        print(f"  ✓ Loaded FluxToAny from {path.name} (v{self.flx_version})")
    
    def _init_empty(self):
        """Initialize empty (for training from scratch)."""
        self.flx_version = '2.0-any'
        self.cse = None
        self.field = None
    
    def _init_text_adapters(self):
        """Initialize text adapters using loaded CSE."""
        from text_adapters import TextToWave, WaveToText
        
        self.input_adapters['text'] = TextToWave(
            cse=self.cse,
            device=self.device,
        )
        
        self.output_adapters['text'] = WaveToText(
            device=self.device,
        )
    
    def add_adapter(
        self,
        modality: str,
        direction: str,  # 'input' or 'output'
        adapter: Optional[Union[XToWave, WaveToX]] = None,
        **kwargs,
    ):
        """
        Add or replace an adapter.
        
        Args:
            modality: 'grid', 'image', 'audio', etc.
            direction: 'input' or 'output'
            adapter: Pre-built adapter, or None to use registry
            **kwargs: Passed to adapter constructor if building
        """
        kwargs['device'] = kwargs.get('device', self.device)
        
        if direction == 'input':
            if adapter:
                self.input_adapters[modality] = adapter
            else:
                self.input_adapters[modality] = get_input_adapter(modality, **kwargs)
        else:
            if adapter:
                self.output_adapters[modality] = adapter
            else:
                self.output_adapters[modality] = get_output_adapter(modality, **kwargs)
        
        print(f"  ✓ Added {direction} adapter: {modality}")
    
    def encode(
        self,
        x: Any,
        modality: str = 'text',
        **kwargs,
    ) -> Tensor:
        """
        Encode input to wave space.
        
        Args:
            x: Input in native format
            modality: Input modality ('text', 'grid', etc.)
        
        Returns:
            [432] or [seq, 432] wave tensor
        """
        if modality not in self.input_adapters:
            raise ValueError(f"No input adapter for: {modality}. Add with add_adapter()")
        
        return self.input_adapters[modality].encode(x, **kwargs)
    
    def decode(
        self,
        wave: Tensor,
        modality: str = 'text',
        **kwargs,
    ) -> Any:
        """
        Decode wave to output modality.
        
        Args:
            wave: [432] or [seq, 432] wave tensor
            modality: Output modality
        
        Returns:
            Output in native format
        """
        if modality not in self.output_adapters:
            raise ValueError(f"No output adapter for: {modality}. Add with add_adapter()")
        
        return self.output_adapters[modality].decode(wave, **kwargs)
    
    def query_field(
        self,
        wave: Tensor,
        k: int = 4,
    ) -> Tuple[Tensor, Tensor, Tensor]:
        """
        Query the resonance field.
        
        Returns:
            (features, similarities, locations) from field
        """
        if self.field is None:
            raise RuntimeError("No field loaded")
        
        wave_vec = wave.mean(dim=0) if wave.dim() > 1 else wave
        return self.field.query(wave_vec, k=k)
    
    def process(
        self,
        x: Any,
        input_modality: str = 'text',
        output_modality: str = 'text',
        use_field: bool = True,
        **decode_kwargs,
    ) -> Any:
        """
        Full processing pipeline: input → encode → field → decode → output.
        
        Args:
            x: Input in native format
            input_modality: 'text', 'grid', etc.
            output_modality: 'text', 'grid', etc.
            use_field: Whether to query field for context
            **decode_kwargs: Passed to output decoder
        
        Returns:
            Output in native format
        """
        # Encode
        wave = self.encode(x, modality=input_modality)
        
        # Query field (optional)
        if use_field and self.field is not None:
            field_features, _, _ = self.query_field(wave)
            context_wave = field_features.mean(dim=0)
            # Blend input with context
            wave_vec = wave.mean(dim=0) if wave.dim() > 1 else wave
            wave_vec = 0.7 * wave_vec + 0.3 * context_wave
        else:
            wave_vec = wave.mean(dim=0) if wave.dim() > 1 else wave
        
        # Decode
        return self.decode(wave_vec, modality=output_modality, **decode_kwargs)
    
    def extract_transform(
        self,
        input_grid: Any,
        output_grid: Any,
    ) -> Tensor:
        """
        Extract transformation rule from input/output pair.
        
        This is the key ARC operation: learn the delta.
        
        Returns:
            [432] delta wave representing the transformation
        """
        if 'grid' not in self.input_adapters:
            self.add_adapter('grid', 'input')
        
        encoder = self.input_adapters['grid']
        _, _, delta = encoder.encode_pair(input_grid, output_grid)
        return delta
    
    def apply_transform(
        self,
        delta_wave: Tensor,
        input_grid: Any,
        grid_size: Optional[Tuple[int, int]] = None,
    ) -> Any:
        """
        Apply extracted transformation to new input.
        
        Args:
            delta_wave: [432] transformation wave
            input_grid: Grid to transform
            grid_size: Output size (if known)
        
        Returns:
            Transformed grid
        """
        if 'grid' not in self.output_adapters:
            self.add_adapter('grid', 'output')
        
        decoder = self.output_adapters['grid']
        return decoder.apply_transform(delta_wave, input_grid, grid_size)
    
    def solve_arc_task(
        self,
        train_pairs: list,
        test_input: Any,
    ) -> Any:
        """
        Solve an ARC task given training examples.
        
        Args:
            train_pairs: List of (input_grid, output_grid) tuples
            test_input: The test input to transform
        
        Returns:
            Predicted output grid
        """
        # Ensure grid adapters are loaded
        if 'grid' not in self.input_adapters:
            self.add_adapter('grid', 'input')
        if 'grid' not in self.output_adapters:
            self.add_adapter('grid', 'output')
        
        # Extract deltas from all training pairs
        deltas = []
        for inp, out in train_pairs:
            delta = self.extract_transform(inp, out)
            deltas.append(delta)
        
        # Average the deltas (the common transformation)
        avg_delta = torch.stack(deltas).mean(dim=0)
        
        # Store in field for future (thermodynamic learning)
        if self.field is not None:
            self.field.perturb(avg_delta)
        
        # Apply to test input
        return self.apply_transform(avg_delta, test_input)
    
    def save_flx(self, path: Path):
        """Save to .flx v2.0 format."""
        flx = {
            'format': 'FLUX',
            'version': '2.0-any',
            'metadata': {
                'adapters': list(self.input_adapters.keys()) + list(self.output_adapters.keys()),
            },
            
            # Core components
            'cse': {
                'config': self.cse.config if hasattr(self.cse, 'config') else {},
                'state_dict': self.cse.state_dict(),
            },
        }
        
        if self.field is not None:
            flx['field'] = {
                'config': getattr(self.field, 'config', {}),
                'state_dict': self.field.state_dict(),
            }
        
        # Save adapters
        flx['adapters'] = {}
        for name, adapter in self.input_adapters.items():
            if name != 'text':  # Text uses CSE, don't duplicate
                flx['adapters'][f'input_{name}'] = {
                    'config': getattr(adapter, 'config', {}),
                    'state_dict': adapter.state_dict(),
                }
        
        for name, adapter in self.output_adapters.items():
            flx['adapters'][f'output_{name}'] = {
                'config': getattr(adapter, 'config', {}),
                'state_dict': adapter.state_dict(),
            }
        
        torch.save(flx, path)
        size_mb = path.stat().st_size / (1024 * 1024)
        print(f"  ✓ Saved Flux-to-any.flx ({size_mb:.1f} MB)")
    
    @classmethod
    def from_flx(cls, path: str, device: str = 'cpu') -> 'FluxToAny':
        """Load from .flx file."""
        return cls(flx_path=Path(path), device=device)
    
    @classmethod 
    def from_beta(cls, device: str = 'cpu') -> 'FluxToAny':
        """Load from Flux-beta.flx."""
        return cls.from_flx('checkpoints/Flux-beta.flx', device=device)
```

## Training the Grid Adapters

```python
"""
train_grid_adapters.py — Train GridToWave/WaveToGrid on ARC data

Uses ARC training set to learn:
1. How to encode grids into meaningful wave representations
2. How to decode waves back to grids
3. How to extract and apply transformations
"""

import torch
import torch.nn.functional as F
from torch.utils.data import DataLoader
from pathlib import Path
import json

from flux_to_any import FluxToAny
from grid_adapters import GridToWave, WaveToGrid


class ARCDataset:
    """Load ARC training data."""
    
    def __init__(self, arc_path: Path):
        self.tasks = []
        for task_file in arc_path.glob('*.json'):
            with open(task_file) as f:
                task = json.load(f)
                self.tasks.append(task)
    
    def __len__(self):
        return len(self.tasks)
    
    def __getitem__(self, idx):
        task = self.tasks[idx]
        train_pairs = [(torch.tensor(ex['input']), torch.tensor(ex['output']))
                       for ex in task['train']]
        test_pairs = [(torch.tensor(ex['input']), torch.tensor(ex.get('output', ex['input'])))
                      for ex in task['test']]
        return train_pairs, test_pairs


def train_grid_adapters(
    model: FluxToAny,
    arc_path: Path,
    epochs: int = 50,
    lr: float = 1e-3,
    device: str = 'cpu',
):
    """
    Train grid adapters on ARC data.
    
    Losses:
    1. Reconstruction: grid → wave → grid should match
    2. Delta consistency: delta from pair 1 ≈ delta from pair 2 (same task)
    3. Transform accuracy: apply(delta, input) ≈ output
    """
    # Ensure adapters exist
    model.add_adapter('grid', 'input')
    model.add_adapter('grid', 'output')
    
    encoder = model.input_adapters['grid']
    decoder = model.output_adapters['grid']
    
    optimizer = torch.optim.AdamW(
        list(encoder.parameters()) + list(decoder.parameters()),
        lr=lr,
    )
    
    dataset = ARCDataset(arc_path)
    
    for epoch in range(epochs):
        total_loss = 0.0
        
        for train_pairs, _ in dataset:
            optimizer.zero_grad()
            
            # Loss 1: Reconstruction
            recon_loss = 0.0
            for inp, out in train_pairs:
                inp = inp.to(device)
                out = out.to(device)
                
                # Encode and decode input
                inp_wave = encoder.encode(inp, mode='holistic')
                inp_recon = decoder.decode(inp_wave, grid_size=inp.shape)
                recon_loss += F.cross_entropy(
                    F.one_hot(inp_recon.long(), 10).float().permute(2, 0, 1).unsqueeze(0),
                    inp.unsqueeze(0),
                )
            
            # Loss 2: Delta consistency
            if len(train_pairs) >= 2:
                deltas = []
                for inp, out in train_pairs:
                    _, _, delta = encoder.encode_pair(inp.to(device), out.to(device))
                    deltas.append(delta)
                
                # All deltas should be similar
                delta_stack = torch.stack(deltas)
                delta_mean = delta_stack.mean(dim=0)
                consistency_loss = F.mse_loss(delta_stack, delta_mean.unsqueeze(0).expand_as(delta_stack))
            else:
                consistency_loss = torch.tensor(0.0, device=device)
            
            # Loss 3: Transform accuracy
            transform_loss = 0.0
            avg_delta = delta_mean if len(train_pairs) >= 2 else deltas[0]
            for inp, out in train_pairs:
                pred_out = decoder.apply_transform(
                    avg_delta,
                    inp.to(device),
                    grid_size=out.shape,
                )
                transform_loss += F.mse_loss(pred_out.float(), out.float().to(device))
            
            # Total loss
            loss = recon_loss + consistency_loss + transform_loss
            loss.backward()
            optimizer.step()
            
            total_loss += loss.item()
        
        print(f"  Epoch {epoch+1}/{epochs}: loss={total_loss/len(dataset):.4f}")
    
    print("  ✓ Grid adapters trained")
```

Phase 8.5 provides the **I/O layer**. Phase 9 provides the **generation engine**.

```
Phase 8.5: FluxToAny

├── Core:   Field + GR + TL + CGN (from Flux-beta)
└── Output: WaveToX adapters (text, grid, image, audio)

Phase 9: WaveGenerator
├── Input:  Uses XToWave from Phase 8.5
├── Core:   Wave-level autoregressive generation
│           wave[t] → field query → wave[t+1]
└── Output: Uses WaveToX from Phase 8.5

    Any modality → XToWave → WaveGenerator → WaveToX → Any modality
```

Phase 9 doesn't need to know about modalities. It just generates waves.
