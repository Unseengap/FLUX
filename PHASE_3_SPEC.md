## gravity.py — Build This First

```python
"""
GravitationalRelevance: Replaces multi-head self-attention.

Core idea:
    Concepts with more accumulated evidence become "heavier".
    New queries naturally fall toward the heaviest relevant concepts.
    No all-pairs comparison needed — spatial index finds neighbors.

Complexity:
    Traditional attention:    O(n²) — every token attends to every token
    Gravitational relevance:  O(log n) — spatial tree finds k neighbors

Key insight from physics:
    Gravity follows inverse square law: F = Gm₁m₂ / r²
    Strong evidence (high mass) at close semantic distance = high pull
    Contradicted concepts (negative mass) actively repel queries
"""

import torch
import torch.nn as nn
import torch.nn.functional as F
from torch import Tensor
from typing import List, Tuple, Optional
import numpy as np

from mass_tracker import MassTracker
from spatial_index import SpatialIndex


class GravitationalRelevance(nn.Module):
    """
    Physics-based replacement for multi-head attention.
    
    Instead of:
        attention(Q, K, V) = softmax(QK^T / sqrt(d)) * V
    
    We have:
        For each query q:
            Find k nearest concept-attractors via spatial index
            Weight each by: mass / distance²
            Return weighted combination of their value vectors
    
    This is O(log n) because the spatial index (KD-tree) finds
    nearest neighbors in log time, not linear time.
    """
    
    def __init__(
        self,
        feature_dim: int = 512,
        k_neighbors: int = 32,
        base_mass: float = 1.0,
        mass_growth_rate: float = 0.01,
        negative_mass_rate: float = 0.05,
        device: str = 'cuda'
    ):
        super().__init__()
        self.feature_dim = feature_dim
        self.k_neighbors = k_neighbors
        self.base_mass = base_mass
        self.device = device
        
        # Mass tracker: maintains evidence mass per concept
        self.mass_tracker = MassTracker(
            feature_dim=feature_dim,
            base_mass=base_mass,
            growth_rate=mass_growth_rate,
            negative_rate=negative_mass_rate,
            device=device
        )
        
        # Spatial index for O(log n) neighbor search
        self.spatial_index = SpatialIndex(
            feature_dim=feature_dim,
            device=device
        )
        
        # Query, Key, Value projections
        # (kept small since we're replacing not augmenting attention)
        self.query_proj = nn.Linear(feature_dim, feature_dim)
        self.key_proj   = nn.Linear(feature_dim, feature_dim)
        self.value_proj = nn.Linear(feature_dim, feature_dim)
        self.out_proj   = nn.Linear(feature_dim, feature_dim)
        
        # Gravity constant (learned, starts at 1.0)
        self.G = nn.Parameter(torch.ones(1))
    
    def gravitational_weight(
        self,
        query: Tensor,
        neighbor_keys: Tensor,
        neighbor_masses: Tensor
    ) -> Tensor:
        """
        Compute gravitational pull of each neighbor on the query.
        
        Physics: F = G * m / r²
        
        Args:
            query:           [feature_dim] query vector
            neighbor_keys:   [k, feature_dim] neighbor key vectors
            neighbor_masses: [k] evidence mass of each neighbor
        Returns:
            [k] gravitational weights (normalized to sum to 1)
        """
        # Compute semantic distances (1 - cosine_similarity for direction)
        # Then convert to spatial distance in feature space
        query_expanded = query.unsqueeze(0).expand_as(neighbor_keys)
        cosine_sim = F.cosine_similarity(query_expanded, neighbor_keys, dim=-1)
        
        # Distance: 0 = identical, 2 = opposite
        distances = 1.0 - cosine_sim + 1e-8  # Avoid division by zero
        
        # Gravitational force: F = G * m / r²
        # Negative mass gives negative force (repulsion)
        forces = self.G * neighbor_masses / (distances ** 2)
        
        # Softmax over positive forces only for attention weights
        # Negative force concepts are explicitly excluded from aggregation
        positive_mask = forces > 0
        forces_masked = torch.where(positive_mask, forces, torch.zeros_like(forces))
        
        if forces_masked.sum() < 1e-8:
            # Edge case: all neighbors repel — return uniform over positives
            weights = torch.ones(len(forces), device=self.device) / len(forces)
        else:
            weights = F.softmax(forces_masked, dim=0)
        
        return weights
    
    def forward(
        self,
        queries: Tensor,
        context: Optional[Tensor] = None
    ) -> Tensor:
        """
        Gravitational relevance forward pass.
        
        For each query position, find k gravitationally nearest concepts
        and return their mass-weighted value combination.
        
        Args:
            queries: [seq_len, feature_dim] or [batch, seq_len, feature_dim]
            context: optional [ctx_len, feature_dim] additional context keys
        Returns:
            [seq_len, feature_dim] attended output (same shape as queries)
        """
        # Handle batch dimension
        batched = queries.dim() == 3
        if batched:
            batch_size, seq_len, _ = queries.shape
            queries = queries.view(-1, self.feature_dim)
        else:
            seq_len = queries.shape[0]
        
        # Project queries
        Q = self.query_proj(queries)  # [seq_len, feature_dim]
        
        # Build key/value context: self-context + any additional context
        if context is not None:
            K_self = self.key_proj(queries)
            K_ctx  = self.key_proj(context)
            K = torch.cat([K_self, K_ctx], dim=0)
            V_self = self.value_proj(queries)
            V_ctx  = self.value_proj(context)
            V = torch.cat([V_self, V_ctx], dim=0)
        else:
            K = self.key_proj(queries)
            V = self.value_proj(queries)
        
        # Register keys with spatial index and mass tracker
        self.spatial_index.add(K.detach())
        self.mass_tracker.observe(K.detach())
        
        # For each query: find k nearest, apply gravity, aggregate values
        outputs = []
        for i in range(len(Q)):
            q = Q[i]
            
            # O(log n) neighbor search via spatial index
            neighbor_indices = self.spatial_index.search(q, k=self.k_neighbors)
            neighbor_keys    = K[neighbor_indices]
            neighbor_values  = V[neighbor_indices]
            neighbor_masses  = self.mass_tracker.get_masses(neighbor_indices)
            
            # Gravitational weighting
            weights = self.gravitational_weight(q, neighbor_keys, neighbor_masses)
            # weights: [k]
            
            # Weighted aggregation of values
            attended = (weights.unsqueeze(-1) * neighbor_values).sum(dim=0)
            outputs.append(attended)
        
        output = torch.stack(outputs, dim=0)  # [seq_len, feature_dim]
        output = self.out_proj(output)
        
        if batched:
            output = output.view(batch_size, seq_len, self.feature_dim)
        
        return output
    
    def reinforce(self, concept_vector: Tensor, strength: float = 1.0):
        """
        Increase mass of a concept — called when evidence confirms it.
        This makes the concept exert stronger gravitational pull on future queries.
        """
        self.mass_tracker.reinforce(concept_vector, strength)
    
    def contradict(self, concept_vector: Tensor, strength: float = 1.0):
        """
        Decrease mass of a concept toward negative — called when evidence contradicts it.
        Negative mass = repulsion. The model physically avoids wrong answers.
        """
        self.mass_tracker.contradict(concept_vector, strength)
    
    def save_state(self) -> dict:
        """Serialize GR state for checkpoint."""
        return {
            'model_state': self.state_dict(),
            'mass_state': self.mass_tracker.save_state(),
            'spatial_index_state': self.spatial_index.save_state(),
            'config': {
                'feature_dim': self.feature_dim,
                'k_neighbors': self.k_neighbors,
                'base_mass': self.base_mass,
            }
        }
    
    @classmethod
    def load_state(cls, state: dict, device: str = 'cuda') -> 'GravitationalRelevance':
        """Restore GR from checkpoint."""
        gr = cls(**state['config'], device=device)
        gr.load_state_dict(state['model_state'])
        gr.mass_tracker.load_state(state['mass_state'])
        gr.spatial_index.load_state(state['spatial_index_state'])
        return gr
```

## mass_tracker.py — Build This Second

```python
"""
MassTracker: Tracks evidence mass for each concept the model encounters.

Mass grows when a concept is repeatedly confirmed.
Mass decreases (toward negative) when a concept is contradicted.
Negative mass = repulsion in the gravitational relevance system.

This is how the model:
- Naturally weighs well-evidenced concepts more heavily
- Automatically deprioritizes concepts that were wrong
- Builds confidence over time without explicit calibration training
"""

import torch
from torch import Tensor
import torch.nn.functional as F
from typing import Optional
import numpy as np


class MassTracker:
    """
    Maintains a registry of concept vectors and their evidence masses.
    
    Concepts are added dynamically as the model observes new keys.
    Similar concepts (cosine sim > threshold) are merged rather than
    stored separately — prevents unbounded memory growth.
    """
    
    def __init__(
        self,
        feature_dim: int = 512,
        base_mass: float = 1.0,
        growth_rate: float = 0.01,
        negative_rate: float = 0.05,
        merge_threshold: float = 0.95,
        max_concepts: int = 100000,
        device: str = 'cuda'
    ):
        self.feature_dim = feature_dim
        self.base_mass = base_mass
        self.growth_rate = growth_rate
        self.negative_rate = negative_rate
        self.merge_threshold = merge_threshold
        self.max_concepts = max_concepts
        self.device = device
        
        # Storage: concept vectors and their masses
        # Pre-allocated for speed, tracked by count
        self.concept_vectors = torch.zeros(
            max_concepts, feature_dim, device=device
        )
        self.masses = torch.full(
            (max_concepts,), base_mass, device=device
        )
        self.count = 0
    
    def observe(self, keys: Tensor):
        """
        Register observed keys and increment their mass.
        Called every forward pass with the current batch of keys.
        
        Args:
            keys: [n, feature_dim] key vectors from current context
        """
        for key in keys:
            self._observe_one(key)
    
    def _observe_one(self, key: Tensor):
        """
        Register a single concept observation.
        If similar concept exists: reinforce it.
        If new: add to registry with base mass.
        """
        if self.count == 0:
            self._add_concept(key)
            return
        
        # Check if similar concept already exists
        existing = self.concept_vectors[:self.count]
        key_expanded = key.unsqueeze(0).expand(self.count, -1)
        similarities = F.cosine_similarity(key_expanded, existing, dim=-1)
        
        max_sim, max_idx = similarities.max(dim=0)
        
        if max_sim.item() > self.merge_threshold:
            # Reinforce existing concept
            self.masses[max_idx] += self.growth_rate
        else:
            # New concept — add to registry
            self._add_concept(key)
    
    def _add_concept(self, vector: Tensor):
        """Add a new concept to the registry."""
        if self.count >= self.max_concepts:
            # Prune lowest-mass concepts when full
            self._prune()
        idx = self.count
        self.concept_vectors[idx] = vector.detach()
        self.masses[idx] = self.base_mass
        self.count += 1
    
    def get_masses(self, indices: Tensor) -> Tensor:
        """
        Get masses for a set of concept indices.
        Used by GravitationalRelevance during forward pass.
        
        Args:
            indices: [k] integer indices into concept registry
        Returns:
            [k] mass values (can be negative for contradicted concepts)
        """
        # Clamp indices to valid range
        valid_indices = indices.clamp(0, max(0, self.count - 1))
        return self.masses[valid_indices]
    
    def reinforce(self, concept: Tensor, strength: float = 1.0):
        """
        Increase mass of closest concept to the given vector.
        Called when external evidence confirms a concept.
        """
        if self.count == 0:
            return
        existing = self.concept_vectors[:self.count]
        concept_expanded = concept.unsqueeze(0).expand(self.count, -1)
        sims = F.cosine_similarity(concept_expanded, existing, dim=-1)
        idx = sims.argmax()
        self.masses[idx] += self.growth_rate * strength
    
    def contradict(self, concept: Tensor, strength: float = 1.0):
        """
        Decrease mass of closest concept — can go negative (repulsion).
        Called when evidence contradicts a concept.
        Negative mass makes this concept repel future similar queries.
        """
        if self.count == 0:
            return
        existing = self.concept_vectors[:self.count]
        concept_expanded = concept.unsqueeze(0).expand(self.count, -1)
        sims = F.cosine_similarity(concept_expanded, existing, dim=-1)
        idx = sims.argmax()
        self.masses[idx] -= self.negative_rate * strength
        # No floor — mass can go arbitrarily negative for strong contradictions
    
    def _prune(self, keep_fraction: float = 0.8):
        """Remove lowest-mass concepts when registry is full."""
        keep_n = int(self.max_concepts * keep_fraction)
        top_indices = self.masses[:self.count].topk(keep_n).indices
        self.concept_vectors[:keep_n] = self.concept_vectors[top_indices]
        self.masses[:keep_n] = self.masses[top_indices]
        self.count = keep_n
    
    def stats(self) -> dict:
        """Return mass distribution statistics."""
        if self.count == 0:
            return {'count': 0}
        active_masses = self.masses[:self.count]
        return {
            'count': self.count,
            'mean_mass': active_masses.mean().item(),
            'max_mass': active_masses.max().item(),
            'min_mass': active_masses.min().item(),
            'negative_count': (active_masses < 0).sum().item(),
            'positive_count': (active_masses > 0).sum().item(),
        }
    
    def save_state(self) -> dict:
        return {
            'concept_vectors': self.concept_vectors[:self.count].cpu(),
            'masses': self.masses[:self.count].cpu(),
            'count': self.count,
            'config': {
                'feature_dim': self.feature_dim,
                'base_mass': self.base_mass,
                'growth_rate': self.growth_rate,
                'negative_rate': self.negative_rate,
            }
        }
    
    def load_state(self, state: dict):
        self.count = state['count']
        n = self.count
        self.concept_vectors[:n] = state['concept_vectors'].to(self.device)
        self.masses[:n] = state['masses'].to(self.device)
```

## spatial_index.py — Build This Third

```python
"""
SpatialIndex: O(log n) nearest neighbor search for GravitationalRelevance.

This is what makes GR fast.
Traditional attention: O(n²) — compare query to every key
SpatialIndex:          O(log n) — KD-tree finds k nearest neighbors

For small n (< 1000): uses sklearn KDTree (simple, accurate)
For large n (> 1000): uses FAISS (GPU-accelerated, production scale)

The index is rebuilt periodically as new concepts are added.
Between rebuilds, new vectors are buffered and searched linearly.
"""

import torch
from torch import Tensor
import numpy as np
from typing import Optional

try:
    import faiss
    FAISS_AVAILABLE = True
except ImportError:
    FAISS_AVAILABLE = False
    print("⚠ FAISS not available — falling back to sklearn KDTree")
    print("  Install: pip install faiss-gpu  (or faiss-cpu)")

try:
    from sklearn.neighbors import KDTree
    SKLEARN_AVAILABLE = True
except ImportError:
    SKLEARN_AVAILABLE = False


class SpatialIndex:
    """
    Maintains a spatial index over concept vectors for fast neighbor search.
    
    Automatically selects FAISS (GPU) or sklearn KDTree based on availability
    and dataset size.
    
    Critical property: index is updated INCREMENTALLY.
    Adding new concepts does not require full rebuild every time.
    Buffer of new vectors searched linearly until rebuild threshold hit.
    """
    
    def __init__(
        self,
        feature_dim: int = 512,
        rebuild_threshold: int = 1000,
        device: str = 'cuda'
    ):
        self.feature_dim = feature_dim
        self.rebuild_threshold = rebuild_threshold
        self.device = device
        
        # All indexed vectors (numpy for compatibility)
        self.indexed_vectors = np.zeros((0, feature_dim), dtype=np.float32)
        
        # Buffer for new vectors not yet indexed
        self.buffer = []
        self.buffer_tensors = []
        
        # The actual index (built lazily)
        self._index = None
        self._index_type = None
        self._index_size = 0
        
        self._select_backend()
    
    def _select_backend(self):
        """Choose best available backend."""
        if FAISS_AVAILABLE:
            self._backend = 'faiss'
        elif SKLEARN_AVAILABLE:
            self._backend = 'sklearn'
        else:
            self._backend = 'brute'
            print("⚠ No fast index available — using brute force O(n) search")
            print("  Performance will degrade at large scale")
    
    def add(self, vectors: Tensor):
        """
        Add new vectors to the index.
        Vectors go to buffer first; index rebuilt when buffer exceeds threshold.
        
        Args:
            vectors: [n, feature_dim] vectors to add
        """
        np_vecs = vectors.detach().cpu().numpy().astype(np.float32)
        # Normalize for cosine similarity search
        norms = np.linalg.norm(np_vecs, axis=1, keepdims=True) + 1e-8
        np_vecs = np_vecs / norms
        
        for v in np_vecs:
            self.buffer.append(v)
        
        # Rebuild if buffer too large
        if len(self.buffer) >= self.rebuild_threshold:
            self._rebuild_index()
    
    def search(self, query: Tensor, k: int = 32) -> Tensor:
        """
        Find k nearest neighbors to query vector.
        
        Args:
            query: [feature_dim] query vector
        Returns:
            [k] integer indices into the combined vector store
        """
        q = query.detach().cpu().numpy().astype(np.float32)
        q = q / (np.linalg.norm(q) + 1e-8)
        
        all_vectors = self._get_all_vectors()
        n = len(all_vectors)
        
        if n == 0:
            return torch.zeros(k, dtype=torch.long, device=self.device)
        
        actual_k = min(k, n)
        
        if self._backend == 'faiss' and self._index is not None and self._index_size > 0:
            indices = self._faiss_search(q, actual_k)
        elif self._backend == 'sklearn' and self._index is not None:
            indices = self._sklearn_search(q, actual_k)
        else:
            indices = self._brute_search(q, actual_k, all_vectors)
        
        # Pad to k if fewer than k results
        if len(indices) < k:
            pad = np.zeros(k - len(indices), dtype=np.int64)
            indices = np.concatenate([indices, pad])
        
        return torch.tensor(indices, dtype=torch.long, device=self.device)
    
    def _rebuild_index(self):
        """Rebuild the spatial index from all current vectors."""
        all_vecs = self._get_all_vectors()
        if len(all_vecs) == 0:
            return
        
        if self._backend == 'faiss':
            self._build_faiss(all_vecs)
        elif self._backend == 'sklearn':
            self._build_sklearn(all_vecs)
        
        self.indexed_vectors = all_vecs
        self.buffer = []
        self._index_size = len(all_vecs)
    
    def _get_all_vectors(self) -> np.ndarray:
        """Combine indexed vectors and buffer."""
        if len(self.buffer) == 0:
            return self.indexed_vectors
        buf = np.stack(self.buffer, axis=0)
        if len(self.indexed_vectors) == 0:
            return buf
        return np.concatenate([self.indexed_vectors, buf], axis=0)
    
    def _build_faiss(self, vectors: np.ndarray):
        """Build FAISS IndexFlatIP (inner product = cosine for normalized vecs)."""
        index = faiss.IndexFlatIP(self.feature_dim)
        index.add(vectors)
        self._index = index
        self._index_type = 'faiss'
    
    def _faiss_search(self, query: np.ndarray, k: int) -> np.ndarray:
        _, indices = self._index.search(query.reshape(1, -1), k)
        return indices[0]
    
    def _build_sklearn(self, vectors: np.ndarray):
        self._index = KDTree(vectors, metric='euclidean')
        self._index_type = 'sklearn'
    
    def _sklearn_search(self, query: np.ndarray, k: int) -> np.ndarray:
        _, indices = self._index.query(query.reshape(1, -1), k=k)
        return indices[0]
    
    def _brute_search(
        self, query: np.ndarray, k: int, vectors: np.ndarray
    ) -> np.ndarray:
        """Fallback O(n) brute force search."""
        sims = vectors @ query
        return np.argsort(sims)[::-1][:k]
    
    def save_state(self) -> dict:
        return {
            'indexed_vectors': self.indexed_vectors,
            'buffer': self.buffer,
            'config': {
                'feature_dim': self.feature_dim,
                'rebuild_threshold': self.rebuild_threshold,
            }
        }
    
    def load_state(self, state: dict):
        self.indexed_vectors = state['indexed_vectors']
        self.buffer = state['buffer']
        if len(self.indexed_vectors) > 0:
            self._rebuild_index()
```

## negative_mass.py — Build This Fourth

```python
"""
NegativeMass: Contradiction detection and repulsion mechanism.

When the model encounters evidence that contradicts a known concept,
that concept's mass goes negative — it repels future similar queries.

This implements the model's ability to:
- Know it is wrong about something
- Actively avoid that wrong conclusion
- Update beliefs when contradicted by strong evidence

Example:
    Model learned: "Pluto is a planet" (positive mass)
    New evidence:  "Pluto was reclassified as a dwarf planet"
    Effect:        "Pluto is a planet" concept gets negative mass
    Future queries about Pluto avoid the "planet" attractor
    and fall toward the "dwarf planet" attractor instead.
"""

import torch
from torch import Tensor
import torch.nn.functional as F
from typing import Tuple


class ContradictionDetector:
    """
    Detects when new evidence contradicts existing beliefs.
    
    A contradiction is detected when:
    1. The new evidence is semantically close to an existing concept
    2. But points in an opposing direction in semantic space
    3. And carries sufficient evidence weight to be credible
    """
    
    def __init__(
        self,
        contradiction_threshold: float = -0.3,
        min_evidence_strength: float = 0.5,
        device: str = 'cuda'
    ):
        self.contradiction_threshold = contradiction_threshold
        self.min_evidence_strength = min_evidence_strength
        self.device = device
    
    def check(
        self,
        new_concept: Tensor,
        existing_concept: Tensor,
        evidence_strength: float = 1.0
    ) -> Tuple[bool, float]:
        """
        Check if new_concept contradicts existing_concept.
        
        Returns:
            (is_contradiction: bool, contradiction_strength: float)
        
        Contradiction detected when:
        - Cosine similarity < contradiction_threshold (opposing directions)
        - evidence_strength >= min_evidence_strength
        """
        if evidence_strength < self.min_evidence_strength:
            return False, 0.0
        
        similarity = F.cosine_similarity(
            new_concept.unsqueeze(0),
            existing_concept.unsqueeze(0)
        ).item()
        
        is_contradiction = similarity < self.contradiction_threshold
        strength = abs(similarity) * evidence_strength if is_contradiction else 0.0
        
        return is_contradiction, strength
    
    def find_contradicted_concepts(
        self,
        new_concept: Tensor,
        concept_registry: Tensor,
        masses: Tensor
    ) -> Tensor:
        """
        Find all existing concepts contradicted by new_concept.
        
        Args:
            new_concept:       [feature_dim] incoming concept
            concept_registry:  [n, feature_dim] existing concepts
            masses:            [n] current masses
        Returns:
            [n] contradiction strengths (0 where no contradiction)
        """
        if len(concept_registry) == 0:
            return torch.zeros(0, device=self.device)
        
        new_expanded = new_concept.unsqueeze(0).expand_as(concept_registry)
        similarities = F.cosine_similarity(new_expanded, concept_registry, dim=-1)
        
        # Only contradict concepts with positive mass (not already contradicted)
        positive_mask = masses > 0
        contradiction_mask = (
            (similarities < self.contradiction_threshold) & positive_mask
        )
        
        # Strength proportional to how opposing they are
        strengths = torch.where(
            contradiction_mask,
            -similarities * 0.1,  # More opposing = stronger contradiction
            torch.zeros_like(similarities)
        )
        
        return strengths
```

## sanity_decoder.py — Build This Fifth

```python
"""
SanityDecoder: First text output from FLUX. Added in Phase 3.

This is NOT the final FLUX decoder — that comes in Phase 7.
This is a quick diagnostic decoder that proves:
    "Information survives the CSE → Field → GR pipeline"

Quality target: recognizable text fragments, NOT fluent generation
If output is complete nonsense at Phase 3 → something is fundamentally broken
If output is garbled but recognizable → pipeline is alive, continue

Architecture:
    Field features [feature_dim]
        → Linear projection [feature_dim → 1024]
        → ReLU
        → Linear projection [1024 → 256]  (256 possible byte values)
        → Softmax → byte probabilities
        → argmax → byte sequence
        → decode UTF-8 → text

This will be REPLACED in Phase 7 with the real autoregressive decoder.
Do not over-engineer this — it's a diagnostic tool only.
"""

import torch
import torch.nn as nn
import torch.nn.functional as F
from torch import Tensor
from typing import List


class SanityDecoder(nn.Module):
    """
    Minimal decoder for diagnostic text output in Phase 3.
    Maps field feature vectors back to byte sequences → text.
    
    This is intentionally simple. Its only job is to prove the
    pipeline is alive, not to produce good text.
    """
    
    def __init__(self, feature_dim: int = 512, device: str = 'cuda'):
        super().__init__()
        self.feature_dim = feature_dim
        self.device = device
        
        self.decode_net = nn.Sequential(
            nn.Linear(feature_dim, 1024),
            nn.ReLU(),
            nn.Dropout(0.1),
            nn.Linear(1024, 512),
            nn.ReLU(),
            nn.Linear(512, 256),   # 256 possible byte values
        )
    
    def forward(self, features: Tensor) -> Tensor:
        """
        Args:
            features: [seq_len, feature_dim] field feature vectors
        Returns:
            [seq_len, 256] byte logits
        """
        return self.decode_net(features)
    
    def decode(self, features: Tensor) -> str:
        """
        Decode field features to text string.
        
        Args:
            features: [seq_len, feature_dim]
        Returns:
            Decoded text string
        """
        with torch.no_grad():
            logits = self.forward(features)
            byte_ids = logits.argmax(dim=-1)
            byte_list = byte_ids.cpu().tolist()
            
            # Filter to printable ASCII range for clean output
            # Full UTF-8 decoding comes in Phase 7
            cleaned = [b for b in byte_list if 32 <= b <= 126]
            
            try:
                return bytes(cleaned).decode('ascii', errors='replace')
            except Exception:
                return '[decode error]'
    
    def reconstruct_and_compare(
        self,
        original_text: str,
        features: Tensor
    ) -> dict:
        """
        Attempt reconstruction and return quality metrics.
        Used by demo_phase3_demo3.py and test_phase3_test4.py
        
        Returns dict with:
            original:     the input text
            reconstructed: what the decoder produced
            char_accuracy: character-level accuracy
            recognizable: bool — is output recognizable as input?
        """
        reconstructed = self.decode(features)
        
        # Character accuracy (rough measure)
        min_len = min(len(original_text), len(reconstructed))
        if min_len == 0:
            char_acc = 0.0
        else:
            matches = sum(
                a == b for a, b in 
                zip(original_text[:min_len], reconstructed[:min_len])
            )
            char_acc = matches / len(original_text)
        
        # "Recognizable" = at least 20% of original chars appear in output
        original_chars = set(original_text.lower())
        output_chars = set(reconstructed.lower())
        overlap = len(original_chars & output_chars) / max(len(original_chars), 1)
        recognizable = overlap > 0.2
        
        return {
            'original': original_text,
            'reconstructed': reconstructed,
            'char_accuracy': char_acc,
            'char_overlap': overlap,
            'recognizable': recognizable,
        }


def pipeline_check(
    cse,
    field,
    gr: 'GravitationalRelevance',
    decoder: SanityDecoder,
    test_input: str,
    verbose: bool = True
) -> dict:
    """
    Run the full CSE → Field → GR → Decoder pipeline on test_input.
    
    This function is called at the end of every Phase 3 demo and test
    to prove the full pipeline is working end to end.
    
    Args:
        cse:        ContinuousSemanticEncoder (Phase 1)
        field:      ResonanceField (Phase 2)
        gr:         GravitationalRelevance (Phase 3)
        decoder:    SanityDecoder (Phase 3)
        test_input: string to encode and decode
        verbose:    print results
    Returns:
        dict with pipeline metrics
    """
    # Step 1: Encode with CSE
    wave = cse.encode(test_input)
    wave_tensor = wave.full  # [seq_len, 432]
    
    # Step 2: Project wave to field feature space
    # (Using field's wave_to_feature projection)
    wave_mean = wave.to_retrieval_vector()  # [432] — mean semantic vector
    field_features_query = field.wave_to_feature(wave_mean)  # [512]
    
    # Step 3: Apply gravitational relevance
    # Query the field for relevant stored features
    field_out = field.query(wave_mean, k=8)  # [8, 512]
    gr_output = gr(field_out.unsqueeze(0)).squeeze(0)  # [8, 512]
    
    # Step 4: Decode to text
    result = decoder.reconstruct_and_compare(test_input, gr_output)
    
    if verbose:
        print(f"\n{'='*50}")
        print(f"PIPELINE CHECK")
        print(f"{'='*50}")
        print(f"Input:         '{test_input}'")
        print(f"Reconstructed: '{result['reconstructed']}'")
        print(f"Char accuracy: {result['char_accuracy']:.1%}")
        print(f"Recognizable:  {'YES ✓' if result['recognizable'] else 'NO ✗'}")
        print(f"{'='*50}\n")
    
    return result
```

## benchmark_attention.py — Build This Sixth

```python
"""
Benchmark: GravitationalRelevance vs PyTorch Multi-Head Attention

Measures wall-clock time and memory usage at multiple sequence lengths.
GR should be faster at seq_len > 512. At seq_len > 4096, attention may OOM.

Run: python benchmark_attention.py

Expected output:
    seq_len=128:    GR: Xms  Attention: Xms  (parity acceptable)
    seq_len=512:    GR: Xms  Attention: Xms  (GR should be close)
    seq_len=1024:   GR: Xms  Attention: Xms  (GR should win)
    seq_len=4096:   GR: Xms  Attention: Xms  (GR should win decisively)
    seq_len=16384:  GR: Xms  Attention: OOM  (GR runs, attention fails)
"""

import torch
import torch.nn as nn
import time
from typing import List, Tuple

SEQ_LENGTHS = [128, 256, 512, 1024, 2048, 4096, 8192, 16384]
FEATURE_DIM = 512
N_HEADS = 8
WARMUP_RUNS = 3
TIMED_RUNS = 10


def benchmark_gr(gr, seq_len: int, device: str) -> Tuple[float, float]:
    """
    Benchmark GR forward pass.
    Returns (mean_ms, peak_memory_mb)
    """
    x = torch.randn(1, seq_len, FEATURE_DIM, device=device)
    
    # Warmup
    for _ in range(WARMUP_RUNS):
        _ = gr(x)
    
    torch.cuda.synchronize() if device == 'cuda' else None
    
    if device == 'cuda':
        torch.cuda.reset_peak_memory_stats()
    
    times = []
    for _ in range(TIMED_RUNS):
        start = time.perf_counter()
        _ = gr(x)
        torch.cuda.synchronize() if device == 'cuda' else None
        end = time.perf_counter()
        times.append((end - start) * 1000)
    
    mean_ms = sum(times) / len(times)
    peak_mb = (
        torch.cuda.max_memory_allocated() / 1e6 
        if device == 'cuda' else 0.0
    )
    return mean_ms, peak_mb


def benchmark_attention(seq_len: int, device: str) -> Tuple[float, float]:
    """
    Benchmark PyTorch MHA forward pass.
    Returns (mean_ms, peak_memory_mb) or ('OOM', 0) on memory error
    """
    mha = nn.MultiheadAttention(
        embed_dim=FEATURE_DIM, 
        num_heads=N_HEADS,
        batch_first=True
    ).to(device)
    
    try:
        x = torch.randn(1, seq_len, FEATURE_DIM, device=device)
        
        for _ in range(WARMUP_RUNS):
            _ = mha(x, x, x)
        
        if device == 'cuda':
            torch.cuda.reset_peak_memory_stats()
        
        times = []
        for _ in range(TIMED_RUNS):
            start = time.perf_counter()
            _ = mha(x, x, x)
            torch.cuda.synchronize() if device == 'cuda' else None
            end = time.perf_counter()
            times.append((end - start) * 1000)
        
        mean_ms = sum(times) / len(times)
        peak_mb = (
            torch.cuda.max_memory_allocated() / 1e6 
            if device == 'cuda' else 0.0
        )
        return mean_ms, peak_mb
    
    except RuntimeError as e:
        if 'out of memory' in str(e).lower():
            return 'OOM', 0.0
        raise


def run_benchmark(gr, device: str = 'cuda') -> List[dict]:
    """Run full benchmark suite and return results."""
    results = []
    
    print(f"\n{'='*65}")
    print(f"{'Seq Len':>10} {'GR (ms)':>12} {'Attn (ms)':>12} {'Speedup':>10} {'GR VRAM':>10}")
    print(f"{'='*65}")
    
    for seq_len in SEQ_LENGTHS:
        gr_ms, gr_mem = benchmark_gr(gr, seq_len, device)
        attn_ms, attn_mem = benchmark_attention(seq_len, device)
        
        if attn_ms == 'OOM':
            speedup = '∞ (OOM)'
            speedup_val = float('inf')
        else:
            speedup_val = attn_ms / gr_ms
            speedup = f'{speedup_val:.1f}x'
        
        attn_str = 'OOM' if attn_ms == 'OOM' else f'{attn_ms:.1f}'
        print(f"{seq_len:>10} {gr_ms:>11.1f} {attn_str:>12} {speedup:>10} {gr_mem:>8.0f}MB")
        
        results.append({
            'seq_len': seq_len,
            'gr_ms': gr_ms,
            'gr_memory_mb': gr_mem,
            'attention_ms': attn_ms,
            'attention_memory_mb': attn_mem,
            'speedup': speedup_val,
        })
    
    print(f"{'='*65}\n")
    return results
```
